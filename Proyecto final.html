# Calculamos el porcentaje de transacciones fraudulentas
def calculoporcentaje(dato, columna):
    contador_transacciones_fraudes = (dato[columna] == 1).sum()
    contador_total_transacciones = len(dato)

    if contador_total_transacciones > 0:
        porcentaje_fraude = (contador_transacciones_fraudes / contador_total_transacciones) * 100
    else:
        porcentaje_fraude = 0
        
    return porcentaje_fraude
        
# Mostramos el porcentaje de transacciones fraudulentas

print("El porcentaje de transacciones fraudulentas es: {:.2f}%".format(porcentaje_fraude))
